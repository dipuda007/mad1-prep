3. Add + Edit event — ONE form template for both

Both routes render event_form.html; edit passes the existing event, add passes event=None, and the template pre-fills with value="{{ event.name if event else '' }}". One template, two jobs — clean thing to show off in the viva.

@app.route('/admin/event/add', methods=['GET', 'POST'])
def add_event():
    user = current_user()
    if not user or user.role != 'admin':
        return redirect(url_for('login'))

    if request.method == 'POST':
        event = Event(name=request.form['name'],
                    location=request.form['location'],
                    category=request.form['category'],
                    duration_hours=int(request.form['duration_hours']),
                    available_slots=int(request.form['available_slots']),
                    status=request.form['status'],
                    start_date=request.form['start_date'],
                    end_date=request.form['end_date'],
                    staff_id=int(request.form['staff_id']) if request.form['staff_id'] else None)
        db.session.add(event)
        db.session.commit()
        flash('Event created.')
        return redirect(url_for('admin_dashboard'))

    staff_list = User.query.filter_by(role='staff', approved=True, blacklisted=False).all()
    return render_template('event_form.html', event=None, staff_list=staff_list)

Notice: staff_list fills the "Assign Staff" dropdown — only approved, non-blacklisted staff. That dropdown IS the "assign staff to events" requirement. The if ... else None handles "no staff yet" (NULL foreign key).

Edit is the same shape — load with get_or_404, overwrite fields from the form, commit (no db.session.add needed — the object is already tracked):

@app.route('/admin/event/edit/<int:event_id>', methods=['GET', 'POST'])
def edit_event(event_id):
    user = current_user()
    if not user or user.role != 'admin':
        return redirect(url_for('login'))

    event = Event.query.get_or_404(event_id)
    if request.method == 'POST':
        event.name = request.form['name']
        event.location = request.form['location']
        event.category = request.form['category']
        event.duration_hours = int(request.form['duration_hours'])
        event.available_slots = int(request.form['available_slots'])
        event.status = request.form['status']
        event.start_date = request.form['start_date']
        event.end_date = request.form['end_date']
        event.staff_id = int(request.form['staff_id']) if request.form['staff_id'] else None
        db.session.commit()
        flash('Event updated.')
        return redirect(url_for('admin_dashboard'))

    staff_list = User.query.filter_by(role='staff', approved=True, blacklisted=False).all()
    return render_template('event_form.html', event=event, staff_list=staff_list)

4. Delete event — POST, and children first

@app.route('/admin/event/delete/<int:event_id>', methods=['POST'])
def delete_event(event_id):
    user = current_user()
    if not user or user.role != 'admin':
        return redirect(url_for('login'))

    event = Event.query.get_or_404(event_id)
    for b in event.bookings:            # remove bookings first (they point to this event)
        db.session.delete(b)
    db.session.delete(event)
    db.session.commit()
    flash('Event deleted.')
    return redirect(url_for('admin_dashboard'))

Two viva answers baked in: (1) delete uses POST because GET must never change data — the delete button is a tiny inline form, not a link; (2) bookings are deleted first because their foreign key points at this event — this is manual cascading.

5. Users & staff page: approve + blacklist

@app.route('/admin/users')
def admin_users():
    user = current_user()
    if not user or user.role != 'admin':
        return redirect(url_for('login'))

    attendees = User.query.filter_by(role='user').all()
    staff = User.query.filter_by(role='staff').all()
    return render_template('admin_users.html', attendees=attendees, staff=staff)


@app.route('/admin/approve/<int:user_id>', methods=['POST'])
def approve_staff(user_id):
    user = current_user()
    if not user or user.role != 'admin':
        return redirect(url_for('login'))

    person = User.query.get_or_404(user_id)
    person.approved = True
    db.session.commit()
    flash(person.name + ' approved as staff.')
    return redirect(url_for('admin_users'))


@app.route('/admin/blacklist/<int:user_id>', methods=['POST'])
def toggle_blacklist(user_id):
    user = current_user()
    if not user or user.role != 'admin':
        return redirect(url_for('login'))

    person = User.query.get_or_404(user_id)
    person.blacklisted = not person.blacklisted     # flips True/False
    db.session.commit()
    flash(person.name + (' blacklisted.' if person.blacklisted else ' un-blacklisted.'))
    return redirect(url_for('admin_users'))

admin_users.html: two tables (staff, attendees); Approve button appears only for unapproved staff ({% if not person.approved %}); Blacklist button text flips with state.

6. Search by name or ID

@app.route('/admin/search')
def admin_search():
    user = current_user()
    if not user or user.role != 'admin':
        return redirect(url_for('login'))

    q = request.args.get('q', '')
    events = Event.query.filter(Event.name.ilike('%' + q + '%')).all()
    people = User.query.filter(User.name.ilike('%' + q + '%')).all()
    if q.isdigit():   # searching by ID also works
        event_by_id = Event.query.get(int(q))
        person_by_id = User.query.get(int(q))
        if event_by_id and event_by_id not in events:
            events.append(event_by_id)
        if person_by_id and person_by_id not in people:
            people.append(person_by_id)
    return render_template('admin_search.html', q=q, events=events, people=people)

GET (search should be shareable), ilike for case-insensitive substring, and if the query is a number it also tries it as an ID. Explain that flow and the search questions in the viva answer themselves.
