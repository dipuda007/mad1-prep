The traffic-director route — this is your "role-specific dashboards after login":

@app.route('/')
def index():
    user = current_user()
    if not user:
        return redirect(url_for('login'))
    if user.role == 'admin':
        return redirect(url_for('admin_dashboard'))
    if user.role == 'staff':
        return redirect(url_for('staff_dashboard'))
    return redirect(url_for('user_dashboard'))

Register — note the two requirement-critical lines: duplicate-username check, and approved=(role == 'user') which makes staff unapproved from birth:

@app.route('/register', methods=['GET', 'POST'])
def register():
    if request.method == 'POST':
        username = request.form['username']
        password = request.form['password']
        name = request.form['name']
        role = request.form['role']            # 'user' or 'staff'
        phone = request.form['phone']

        if User.query.filter_by(username=username).first():
            flash('Username already taken, pick another one.')
            return redirect(url_for('register'))

        user = User(username=username,
                    password_hash=generate_password_hash(password),
                    name=name,
                    role=role,
                    approved=(role == 'user'))   # staff must wait for admin approval
        db.session.add(user)
        db.session.commit()

        if role == 'staff':
            profile = StaffProfile(user_id=user.id, phone=phone)
            db.session.add(profile)
            db.session.commit()
            flash('Registered! Wait for admin approval before logging in.')
        else:
            flash('Registered! You can log in now.')
        return redirect(url_for('login'))

    return render_template('register.html')

Why is there no "admin" option in the role dropdown? Because the statement says admin cannot register — admin exists from Build 1's seeding. Say that sentence in the viva.

Login — four gates in order: user exists → password matches → not blacklisted → staff approved:

@app.route('/login', methods=['GET', 'POST'])
def login():
    if request.method == 'POST':
        username = request.form['username']
        password = request.form['password']
        user = User.query.filter_by(username=username).first()

        if not user or not check_password_hash(user.password_hash, password):
            flash('Wrong username or password.')
            return redirect(url_for('login'))
        if user.blacklisted:
            flash('Your account is blacklisted. Contact admin.')
            return redirect(url_for('login'))
        if user.role == 'staff' and not user.approved:
            flash('Your staff account is not approved yet.')
            return redirect(url_for('login'))

        session['user_id'] = user.id
        session['role'] = user.role
        return redirect(url_for('index'))

    return render_template('login.html')


@app.route('/logout')
def logout():
    session.clear()
    flash('Logged out.')
    return redirect(url_for('login'))

Profile (edit own name/password, staff also phone/experience):

@app.route('/profile', methods=['GET', 'POST'])
def profile():
    user = current_user()
    if not user:
        return redirect(url_for('login'))

    if request.method == 'POST':
        user.name = request.form['name']
        if request.form['password']:                       # only change if box not empty
            user.password_hash = generate_password_hash(request.form['password'])
        if user.staff_profile:
            user.staff_profile.phone = request.form['phone']
            user.staff_profile.experience = request.form['experience']
        db.session.commit()
        flash('Profile updated.')
        return redirect(url_for('profile'))

    return render_template('profile.html', user=user)
